📋 Código para copiar y pegar: loop de entrenamiento completo

Este ejemplo construye un dataset sintético, define un modelo sencillo de clasificación binaria y recorre el loop: forward → loss → backward → step, imprimiendo el loss y la exactitud. 

# Dataset de ejemplo: puntos 2D + etiquetas 0/1
N = 200
X_train = torch.randn(N, 2)

w_true = torch.tensor([[2.0, -3.0]]).t()
b_true = torch.tensor([0.5])

logits_true = X_train @ w_true + b_true
y_train = (logits_true > 0).float().unsqueeze(1)

print("Proporción clase 1:", y_train.mean().item())

# Modelo: neurona logística 2 -> 1
modelo = nn.Sequential(
    nn.Linear(2, 1),
    nn.Sigmoid()
)

criterio = nn.BCELoss()
opt = torch.optim.SGD(modelo.parameters(), lr=0.1)

for epoca in range(200):
    # 1. Forward
    y_pred = modelo(X_train)
    loss = criterio(y_pred, y_train)

    # 2. Backward
    opt.zero_grad()
    loss.backward()
    opt.step()

    if epoca % 20 == 0:
        print(f"Epoch {epoca} | Loss = {loss.item():.4f}")

with torch.no_grad():
    probs = modelo(X_train)
    preds = (probs >= 0.5).float()
    acc = (preds == y_train).float().mean().item()

print(f"Exactitud final: {acc:.3f}")
for epoch in range(100):
    optimizer.zero_grad()          # 1. Limpiar gradientes
    predictions = model(X_train)   # 2. Forward pass
    loss = criterion(predictions, y_train)  # 3. Calcular loss
    loss.backward()                # 4. Backward pass
    optimizer.step()               # 5. Actualizar pesos
    
    if epoch % 10 == 0:
        print(f'Epoch {epoch} | Loss: {loss.item():.4f}')
